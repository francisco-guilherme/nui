## About

The Emoji Picker is built on top of [`frimousse`](https://frimousse.liveblocks.io/), which is created by [liveblocks](https://liveblocks.io).

## Usage

```tsx title="Imports"
import {
  EmojiPicker,
  EmojiPickerContent,
  EmojiPickerEmpty,
  EmojiPickerList,
  EmojiPickerLoading,
  EmojiPickerSearch,
  EmojiPickerSkinToneSelector,
} from "@/components/ui/emoji-picker";
```

```tsx title="Anatomy"
<EmojiPicker>
  <EmojiPickerSearch />
  <EmojiPickerSkinToneSelector />
  <EmojiPickerContent>
    <EmojiPickerLoading />
    <EmojiPickerEmpty />
    <EmojiPickerList />
  </EmojiPickerContent>
</EmojiPicker>
```
